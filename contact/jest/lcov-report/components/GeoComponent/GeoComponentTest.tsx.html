  labelsList: Record<string, Array<string>>;

class GeoComponentTest extends React.Component<GeoComponentProps, GeoComponentState>
{
 
  private CUSTOMEVENTNAME: string = "RecommendedContact";
  private fileName: string = "GeoComponent.tsx";
  private dropdownRef = React.createRef<HTMLDivElement>();
  private labelsFullList: Record<string, Array<string>>;
 
  constructor(props: GeoComponentProps) {
    super(props);
 
    this.labelsFullList = props.dropdownData;
 
    this.state = {
      labelsList: this.labelsFullList,
      isMenuVisible: false,
      checkedList: this.setCheckboxesState(),
      searchText: "",
      isGeoRegionExpanded: this.setDropdownState()
    }
 
    this.toggleDropdown = this.toggleDropdown.bind(this);
    this.toggleGeoRegion = this.toggleGeoRegion.bind(this);
    this.toggleGeoCheckbox = this.toggleGeoCheckbox.bind(this);
    this.clearAll = this.clearAll.bind(this);
    this.onApply = this.onApply.bind(this);
    this.handleClickOutside = this.handleClickOutside.bind(this);
    this.geoDropdownSearch = this.geoDropdownSearch.bind(this);
    this.componentDidMount = this.componentDidMount.bind(this);
    this.componentWillUnmount = this.componentWillUnmount.bind(this);

  /* setDropdownState(): Record<string, boolean>
  example model in geo dropdown menu- 
  nextIsExpanded = {
    Asia Pacific Area: false,  // not expanded
    India Area: true          // expanded
  }
  */
  private setDropdownState(): Record<string, boolean> {
    let nextIsExpanded: Record<string, boolean> = {};
    for (let region in this.labelsFullList) {
      nextIsExpanded[region] = false;
    }
    return nextIsExpanded;
  }
 
  /* setCheckboxesState(): Record<string, Record<string, boolean>>
  example model of checkboxes in geo dropdown menu- 
  checkedList = {
    Asia Pacifi Area: {
      Construction: false,  // unchecked
      Education: true       // checked
    }
    India Area: {
      India: false,
      Pakistan: false
    }
  }
  */
  private setCheckboxesState(): Record<string, Record<string, boolean>> {
    let nextCheckedList: Record<string, Record<string, boolean>> = {};
    for (let region in this.labelsFullList) {
      nextCheckedList[region] = {};
    }
    for (let region in this.labelsFullList) {
      let countries = this.labelsFullList[region];
      for (let i = 0; i < countries.length; i++) {
        nextCheckedList[region][countries[i]] = false;
      }
    }
    return nextCheckedList;
  }
 
  /* toggleDropdown()
  onClick of the dropdown buttons (Industry, Role)
  */
  private toggleDropdown(): void {
    try {
      let nextIsMenuVisible = this.state.isMenuVisible;
      nextIsMenuVisible = !nextIsMenuVisible;
      this.setState({
        isMenuVisible: nextIsMenuVisible
      });
      //Logging.AppInsightsTrackEvent(this.CUSTOMEVENTNAME, "Information", "toggleDropdown()", "Dropdown click event in Geo dropdown", "Rc-Geo-dropdown");
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at toggleDropdown() File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* toggleGeoRegion(string)
  onClick of region
  */
  private toggleGeoRegion(region: string): void {
    try {
      let nextIsGeoRegionExpanded = { ...this.state.isGeoRegionExpanded };
      nextIsGeoRegionExpanded[region] = !nextIsGeoRegionExpanded[region];
      this.setState({
        isGeoRegionExpanded: nextIsGeoRegionExpanded
      });
      //Logging.AppInsightsTrackEvent(this.CUSTOMEVENTNAME, "Information", "toggleGeoRegion()", "GeoDropdown " + region + " region click event in Geo dropdown", "Rc-Geo-" + region.replace(/ /g, ''));
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at toggleGeoRegion() region: " + region + " File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* toggleCheckbox(string)
  onClick of the checkbox 
  */
  private toggleGeoCheckbox(region: string, id: string): void {
    try {
      let nextCheckedList = { ...this.state.checkedList };
      nextCheckedList[region][id] = !nextCheckedList[region][id];
      this.setState({
        checkedList: nextCheckedList
      });
      //Logging.AppInsightsTrackEvent(this.CUSTOMEVENTNAME, "Information", "toggleGeoCheckbox()", "Geo dropdown: " + id + " in " + region + " click event", "Rc-Geo-" + region.replace(/ /g, '') + id);
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at toggleGeoCheckbox() args:(" + region + "," + id + ")+File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* clearAll()
  clear all checkboxes and call parent's  clearFilters()
  */
  private clearAll(): void {
    try {
      let nextCheckedList = { ...this.state.checkedList };
      for (let region in nextCheckedList) {
        for (let country in nextCheckedList[region]) {
          nextCheckedList[region][country] = false;
        }
      }
      this.setState({
        checkedList: nextCheckedList
      });
      this.props.clearFilters(this.props.dropdownTitle);
      //Logging.AppInsightsTrackEvent(this.CUSTOMEVENTNAME, "Information", "clearAll()", "GeoDropdown clear button click event", "Rc-Geo-Clear");
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at clearAll() File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* onApply()
  call parent's applyFilters()
  */
  private onApply(): void {
    try {
      this.props.applyFilters(this.state.checkedList);
      //Logging.AppInsightsTrackEvent(this.CUSTOMEVENTNAME, "Information", "onApply()", "GeoDropdown apply button click event", "Rc-Geo-Apply");
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at onApply() File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* componentDidMount()
  mousedown - to listen for click outside dropdown
  keydown - to listen for keypress outside dropdown and close dropdown on 'esc' keypress

  public async componentDidMount() {
    try {
      document.addEventListener("mousedown", this.handleClickOutside);
      document.addEventListener("keydown", this.handleClickOutside);
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at componentDidMount() File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* componentWillUnmount()
  remove listeners set on componentDidMount()
  */
  public async componentWillUnmount() {
    try {
      document.removeEventListener("mousedown", this.handleClickOutside);
      document.removeEventListener("keydown", this.handleClickOutside);
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at componentWillUnmount() File: " + this.fileName, error);
      console.log(error);
    }
  }
 
  /* handleClickOutside(KeyboardEvent | MouseEvent)
  close dropdown if - 
  1. there is a click outside
  2. there is a keypress outside
  3. esc key pressed inside
  */
  private handleClickOutside(event: MouseEvent | KeyboardEvent): void {
    try {
      let nextIsMenuVisible: boolean = this.state.isMenuVisible;
      if (this.state.isMenuVisible && this.dropdownRef.current && (!this.dropdownRef.current.contains(event.target as Node) || (event as KeyboardEvent).keyCode == 27)) {
        nextIsMenuVisible = false;
      }
      this.setState({
        isMenuVisible: nextIsMenuVisible
      });
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at handleClickOutside() File: " + this.fileName, error);
      console.log(error);
    }
  };
 
  /* searchGeoString(string, string)
  search in single label
  */
  private searchGeoString(inputVal: string, searchStr: string): boolean {
    return searchStr.toLowerCase().includes(inputVal.toLowerCase());
  }
 
  /* dropdownSearch(React.ChangeEvent<HTMLInputElement> | undefined)
  search all checkbox labels
  */
  private geoDropdownSearch(event?: React.ChangeEvent<HTMLInputElement>): void {
    try {
      let inputVal: string;
      if (typeof event == 'undefined') {
        inputVal = '';
      }
      else {
        inputVal = event.target.value;
      }
      let nextIsGeoRegionExpanded = { ...this.state.isGeoRegionExpanded };
      let nextLabelsFullList = { ...this.labelsFullList };
      if (inputVal != "") {
        for (let region in nextLabelsFullList) {
          nextIsGeoRegionExpanded[region] = true;
          nextLabelsFullList[region] = this.labelsFullList[region].filter((country: string) => this.searchGeoString(inputVal, country));
          if (nextLabelsFullList[region].length == 0 && this.searchGeoString(inputVal, region)) {
            nextLabelsFullList[region] = this.labelsFullList[region];
            nextIsGeoRegionExpanded[region] = false;
          }
          else if (nextLabelsFullList[region].length == 0) {
            delete nextLabelsFullList[region];
          }
        }
      }
      else {
        for (let region in nextLabelsFullList) {
          nextIsGeoRegionExpanded[region] = false;
        }
      }
      this.setState({
        searchText: inputVal,
        labelsList: nextLabelsFullList,
        isGeoRegionExpanded: nextIsGeoRegionExpanded
      });
    }
    catch (error) {
      //Logging.AppInsightsTrackException(this.CUSTOMEVENTNAME, "Exception occured at geoDropdownSearch() File: " + this.fileName, error);
      console.log(error);
    }
  }

    return (

        <button tabIndex={0} onClick={this.toggleDropdown} className={styles.dropdownButton}>{this.props.dropdownTitle}&nbsp;&nbsp; <Icon iconName="ChevronDown" aria-hidden="true" /></button>

          <div className={styles.clearApply}>
            <DefaultButton className={styles.buttonClearApply} text="Clear" onClick={this.clearAll} />
            <DefaultButton className={`${styles.buttonClearApply} ${styles.applyButton}`} text="Apply" onClick={this.onApply} />
          </div>
        </div>}
      </div>
    );
  }
